  n_increment_cart -> n_cat_loop [label="DONGU: Kategorilerde gezinmeye devam veya Sepeti Gör"];

  // User chooses to view/edit cart
  n_cat_loop -> n_view_cart_loop [label="Kullanıcı -> Sepeti Gör"];

  // View/Edit cart loop: edits, apply coupon
  n_view_cart_loop -> n_apply_discount;
  n_apply_discount -> n_validate_coupon [label="EĞER evet"];
  n_apply_discount -> n_minimum_check [label="EĞER hayır"];

  n_validate_coupon -> n_apply_coupon_success [label="EĞER evet"];
  n_validate_coupon -> n_coupon_invalid [label="EĞER hayır"];
  n_coupon_invalid -> n_view_cart_loop [label="DONGU: Yeniden dene veya devam"];
  n_apply_coupon_success -> n_minimum_check;

  // Minimum order check
  n_minimum_check -> n_below_minimum [label="EĞER hayır"];
  n_minimum_check -> n_shipping_calc [label="EĞER evet"];
  n_below_minimum -> n_view_cart_loop [label="DONGU: Kullanıcı Sepeti Günceller veya Ürün Ekle"];

  // Shipping calculation
  n_shipping_calc -> n_shipping_free [label="EĞER evet"];
  n_shipping_calc -> n_shipping_charge [label="EĞER hayır"];
  n_shipping_free -> n_payment_selection;
  n_shipping_charge -> n_payment_selection;

  // Payment selection loop & validation
  n_payment_selection -> n_payment_method_check;
  n_payment_method_check -> n_payment_process [label="EĞER evet"];
  n_payment_method_check -> n_payment_selection [label="EĞER hayır -> Başka yöntem seç (DONGU)"];

  // Payment processing -> success/fail
  n_payment_process -> n_payment_success;
  n_payment_success -> n_order_confirm [label="EĞER evet"];
  n_payment_success -> n_payment_fail [label="EĞER hayır"];
  n_payment_fail -> n_payment_selection [label="DONGU: Tekrar Ödeme Yöntemi Seçimi"];

  // After order confirmation -> end
  n_order_confirm -> end;

  // Safety loops: kullanıcı istediğinde sepet görüntülemeye geri dönebilir
  n_view_cart_loop -> n_cat_loop [label="Kullanıcı -> Alışverişe devam et (DONGU)"];
  n_view_cart_loop -> n_payment_selection [label="Kullanıcı -> Ödemeye geç"];

  // Aesthetic
  { rank=same; n_shipping_free; n_shipping_charge; }
}
